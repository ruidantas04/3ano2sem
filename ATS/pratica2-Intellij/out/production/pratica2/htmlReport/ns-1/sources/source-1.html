         this.id = id;

         if (i < 0 || i > 20) {
             throw new NotaInvalidaException();
         }
         this.nota.put(ats, i);
     }
 
     public void incrementaNota(String ats, int i) throws NotaInvalidaException{
         nota.put(ats, nota.getOrDefault(ats, 0) + 5);
     }

     }
 
     public double media() {
         if (nota.isEmpty()) {
             return 0.0;
         }
 
         double soma = 0;
         for (int valor : nota.values()) {
             soma += valor;
         }
 
         return soma / nota.size();
     }
